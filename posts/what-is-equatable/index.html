class Shoes {
  final int productId;

  final String brand;
  final int size;

    required this.name,
    required this.brand,
    required this.size,
  });

  @override
  bool operator ==(Object other) =>
      other is Shoes &&
      other.productId == productId &&
      other.name == name &&
      other.brand == brand &&
      other.size == size;

  @override
  int get hashCode => productId;
